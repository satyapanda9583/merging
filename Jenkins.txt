pipeline {
    agent any // Specifies that the pipeline can run on any available agent

    stages {
        stage('Build') {
            steps {
                echo 'Building the application1.0'
                sh 'mvn deployment' // Example: Execute a Maven build command
            }
        }

        stage('Test') {
            steps {
                echo 'Running tests...'
                sh 'mvn test' // Example: Execute Maven tests
                junit 'target/surefire-reports/*.xml' // Archive JUnit test results
            }
        }

        stage('Deploy') {
            steps {
                echo 'Deploying the application...'
                // Example: Copy artifacts to a deployment location
                sh 'scp target/*.jar user@remotehost:/opt/app/'
            }
        }
    }
}
